  height_snps = read.table(paste0("/Users/nicholeyang/Downloads/simulation-calibrated-mr/data/snp_list/height_",
                                          pvals[i], ".txt"), header = FALSE)
  dat1 = dat.exp[dat.exp$variant %in% height_snps$V1, ]

  # IVW
  res.mr = matrix(NA, ncol = 4, nrow = 2)
  colnames(res.mr) = c("calibrated", "raw", "internal.gwas", "external.gwas")
  rownames(res.mr) = c("point.est", "std")
  
  # calibrated
  MRInput <- mr_input(bx = dat1$cali,
                      bxse = sqrt(dat1$cali.var),
                      by = dat2$cali,
                      byse = sqrt(dat2$cali.var))
  mr = mr_ivw(MRInput)
  res.mr[, 1] = c(mr@Estimate, mr@StdError)
  
  
  # uncalibrated
  MRInput <- mr_input(bx = dat1$raw,
                      bxse = sqrt(dat1$raw.var),
                      by = dat2$raw,
                      byse = sqrt(dat2$raw.var))
  mr = mr_ivw(MRInput)
  res.mr[, 2] = c(mr@Estimate, mr@StdError)

  mr = mr_ivw(MRInput)
  res.mr[, 3] = c(mr@Estimate, mr@StdError)
  
  
  # external gwas
  MRInput <- mr_input(bx = dat1$BETA,
                      bxse = dat1$SE,
                      by = dat2$BETA,
                      byse = dat1$SE)
  mr = mr_ivw(MRInput)
  res.mr[, 4] = c(mr@Estimate, mr@StdError)
  print(res.mr)
  
  labels <- c('calibrated', 'raw', 'internal.gwas', 'external.gwas')
  point_est <- res.mr[1, ]
  error_bars <- 1.96 * res.mr[2, ]
  
  # Plotting
  plot(1:4, point_est, ylim=c(min(point_est - error_bars), max(point_est + error_bars)),
       xaxt='n', pch=19, xlab='', ylab='Point Estimate', main=paste0('IVW: height on EduYrs: pval=', pvals[i]) , cex=1.5)
